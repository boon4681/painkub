kub: used by = ws statement ws
statement: used by kub, block = stmt (stmt ws)*
stmt: used by statement = declaration / function_call / if_stmt / while_stmt / for_stmt / assign / return_stmt
assign: used by stmt = ident WS [+\-*\/]?"=" WS (literal / ident / expression)
if_stmt: used by stmt = "if" WS condition ws block
while_stmt: used by stmt = "while" WS condition ws block
for_stmt: used by stmt = "for" WS ident WS "in" WS range_expr ws block
return_stmt: used by stmt = "return" " " expression
function_call: used by stmt = ident "(" ws func_params ws ")"
func_params: used by function_call = (literal / ident) (ws "," ws (literal / ident))*
condition: used by if_stmt, while_stmt = disjunction
disjunction: used by condition = conjunction (ws "or" ws conjunction)*
conjunction: used by disjunction = comparison (ws "and" ws comparison)*
comparison: used by conjunction = expression ws comparison_opr ws expression
comparison_opr: used by comparison = ">" / "<" / "==" / ">=" / "<="
block: used by if_stmt, while_stmt, for_stmt, function_declaration = "{" ws statement ws "}"
expression: used by assign, return_stmt, comparison, parenthesized_expr = additive_expr
range_expr: used by for_stmt = additive_expr ".." additive_expr
parenthesized_expr: used by primary_expr = "(" ws expression ws ")"
additive_expr: used by expression, range_expr = multiplicative_expr (ws additive_opr ws multiplicative_expr)*
multiplicative_expr: used by additive_expr = primary_expr (ws multiplicative_opr ws primary_expr)*
primary_expr: used by multiplicative_expr = parenthesized_expr / literal / ident
additive_opr: used by additive_expr = "+" / "-"
multiplicative_opr: used by multiplicative_expr = "*" / "/"
declaration: used by stmt = function_declaration / variable_declaration
function_declaration: used by declaration = "fun" WS ident ws "(" ws ws ")" ws block
variable_declaration: used by declaration = "let" WS ident ws "=" ws literal / "let" WS ident ws
literal: used by assign, func_params, primary_expr, variable_declaration = number / string
number: used by literal = [+-]? (int / float)
int: used by number, float = [1-9][0-9]+ / [0-9]
float: used by number = int "." [0-9]+
string: used by literal, single_quote_string, double_quote_string = double_quote_string / single_quote_string
single_quote_string: used by string = "'" (escape_string / single_quote_char_string)* "'"
single_quote_char_string: used by single_quote_string = [^\\\t\r\n\"\']
double_quote_string: used by string = '"' (escape_string / double_quote_char_string)* '"'
double_quote_char_string: used by double_quote_string = [^\\\t\r\n\"\']
escape_string: used by single_quote_string, double_quote_string = '\\' [tbrn"'\\]
ident: used by assign, for_stmt, function_call, func_params, primary_expr, function_declaration, variable_declaration = [a-zA-Z_][a-zA-Z_0-9]*
WS: used by assign, if_stmt, while_stmt, for_stmt, function_declaration, variable_declaration "whitespace"
ws: used by kub, statement, if_stmt, while_stmt, for_stmt, function_call, func_params, disjunction, conjunction, comparison, block, parenthesized_expr, additive_expr, multiplicative_expr, function_declaration, variable_declaration "whitespace"
